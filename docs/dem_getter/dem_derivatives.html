  20def get_raster_as_grid(raster):
  21    """Loads in a digital elevation model as a numpy array and converts any no-data to np.nan
  22
  23        Args:
  24            raster (str OR gdal.Dataset): The path to a single band, gdal readable, digital elevation model OR
  25                an already loaded raster dataset
  26
  27        Returns:
  28            rasterGrid (numpy.ndarray): The elevation data stored as a grid
  29            dx (float): The x coordinate spacing of the grid
  30            dy (float): The y coordinate spacing of the grid
  31            zFactor (float): The estimated tranformation between horizontal and vertical coordinates
  32
  33        Raises:
  34            Exception: Input is neither a path to a raster or a gdal.Dataset
  35        """
  36    #If this is a raster dataset
  37    if isinstance(raster,gdal.Dataset):
  38        doClose = False
  39    
  40    #If this is a file
  41    elif os.path.isfile(raster):   
  42        #Get the raster grid
  43        raster = gdal.Open(raster)
  44
  45        #Close this file after the operation completes
  46        doClose = True
  47    else:
  48        raise Exception('Specified raster is neither a path to a raster or a gdal.Dataset. Please specify a valid path.')
  49
  50    rasterGrid = raster.ReadAsArray().astype(float)
  51    NDV = raster.GetRasterBand(1).GetNoDataValue()
  52
  53    #Mask out NDVs as nan
  54    rasterGrid[rasterGrid==NDV] = np.nan
  55
  56    #Estimate the zFactor for this dataset (if necessary)

 142    """Create a copy of the specified raster with the data present in the numpy ndarray 'array',
 143    and save the file with a new suffix in the same directory as the source
 144
 145    Args:
 146        array (numpy.ndarray): Array with the same dimensions as that specified in pathToRaster
 147        templateRaster (str,path OR gdal.Dataset): Path to a raster dataset that can be loaded by gdal, 
 148        OR a dataset that has already been loaded
 149        newFileName (str, path): New path to save the new raster dataset
 150        doReturnGdalSource (bool): Should the gdal.DataSource object be returned (True), or should
 151            the file be closed (False, default)
 152
 153    Returns:
 154        outDataset (gdal.Dataset OR None): If doReturnGdalSource is True, returns the opened gdal.Datasource object for the new
 155        file. Otherwise returns None
 156
 157    Raises:
 158        Exception: Input is neither a path to a raster or a gdal.Dataset
 159    """
 160    #If this is a raster dataset
 161    if isinstance(templateRaster,gdal.Dataset):
 162        doClose = False
 163    
 164    #If this is a file
 165    elif os.path.isfile(templateRaster):   
 166        #Get the raster grid
 167        templateRaster = gdal.Open(templateRaster)
 168
 169        #Close this file after the operation completes
 170        doClose = True
 171    else:
 172        raise Exception('Specified raster is neither a path to a raster or a gdal.Dataset. Please specify a valid path.')
 173
 174    #Hmmm.... would much rather get this dynamically from the raster, but for now just setting to 32 bit float
 175    gdal_datatype = gdal.GDT_Float32
 176
 177    #What driver (e.g., file type) was used?
 178    if (not newFileName) or (newFileName.lower() == 'memory'):
 179        #If no path was specified, get something in memory
 180        drvr = gdal.GetDriverByName("MEM")
 181        doReturnGdalSource = True
 182    else:
 183        #If a path was specified, this is to be saved - get the filetype from the saved file
 184        drvr = templateRaster.GetDriver()
 185
 186        if drvr.GetDescription() =='MEM': #if the user wants the raster saved but a memory file is input,
 187            #try to get the driver from the save path extension
 188            ext=os.path.splitext(os.path.split(newFileName)[-1])[-1].split('.')[-1]
 189            try:
 190                drvr=gdal.GetDriverByName(FILE_EXT_DICT[ext])
 191
 192            except: #if the save path ext isn't in the dict then assign it to GTiff
 193                import warnings
 194                warnings.warn("The requested file format is not available. Saving as GTiff instead.")
 195                drvr=gdal.GetDriverByName("GTiff")
 196
 197    #Get relevant info from the srcDataset
 198    NDV = templateRaster.GetRasterBand(1).GetNoDataValue() #Get the no data value
 199
 200    #Copy the array so we can update any nan values to the NDV
 201    arrayCopy = array.copy()
 202    arrayCopy[np.isnan(arrayCopy)] = NDV
 203
 204    #Create the output dataset
 205    outDataset = drvr.Create(newFileName,templateRaster.RasterXSize,
 206                             templateRaster.RasterYSize,1,gdal_datatype)
 207    
 208    #NOTE! Before I always set the NDV after writing an array... but in Arc, when using 'MEM' driver...
 209    #this was causing all values to be replaced with NDV
 210    outDataset.GetRasterBand(1).SetNoDataValue(NDV) #Write the no data value
 211    outDataset.GetRasterBand(1).WriteArray(arrayCopy) #Write the new array to it
 212
 213    #Calculate the statistics
 214    outDataset.GetRasterBand(1).ComputeStatistics(0)
 215
 216    #Populate the georeferencing info, etc
 217    outDataset.SetGeoTransform(templateRaster.GetGeoTransform())
 218    outDataset.SetProjection(templateRaster.GetProjection())
 219    outDataset.SetSpatialRef(templateRaster.GetSpatialRef())
 220    outDataset.SetMetadata(templateRaster.GetMetadata())
 221
 222    drvr = None
 223
 224    #Clean up file connections
 225    if not(doReturnGdalSource):
 226        outDataset = None
 227
 228    #Close the template raster if it wasn't already loaded
 229    if doClose:
 230        templateRaster = None
 231
 232    return outDataset

 234def estimate_z_factor(rasterSrc:gdal.Dataset):
 235    """Estimate the multiplicative factor to make the dimensions of the elevations of the raster comparable
 236    to the dimensions of the raster's pixels
 237
 238    Args:
 239        rasterSrc (gdal.Dataset): An open gdal raster dataset. For example, rasterSrc = gdal.Open(filename)
 240
 241    Returns:
 242        zFactor (float): A number that represents the conversion necessary from the vertical units of the raster
 243            to the horizontal units defined by the projection. This will typically be 1 (e.g., both elevation measurements
 244            and coordinates are in meters) or something much less than 1 (e.g., elevation measurements are in meters, and
 245            coordinates degrees).
 246    """
 247    #Z factor estimates rely on the table found here:
 248    # https://www.esri.com/arcgis-blog/products/product/imagery/setting-the-z-factor-parameter-correctly/
 249    
 250
 251    #First check if this is a geographic coordinate system, if not (and its projected) assume
 252    #a z_factor of 1
 253    zFactor = 1
 254
 255    #Get the spatial reference of the raster
 256    spatial_ref = rasterSrc.GetSpatialRef()
 257    
 258    #If this is geographic
 259    if spatial_ref.IsGeographic():
 260
 261        #And the angular units are degrees
 262        if spatial_ref.GetAngularUnitsName().lower() == 'degree':
 263            ##Define how to map coordinates to z_factors - necessary for approximating derivatives
 264            #When spatial coordinates are used that don't match elevation coordinates
 265            #This is an array: latitude, meters->degrees, feet -> degrees
 266            zFactorsTable = np.array([
 267            [0,0.00000898, 0.00000273],
 268            [10,0.00000912,	0.00000278],
 269            [20,0.00000956,	0.00000291],
 270            [30,0.00001036,	0.00000316],
 271            [40,0.00001171,	0.00000357],
 272            [50,0.00001395,	0.00000425],
 273            [60,0.00001792,	0.00000546],
 274            [70,0.00002619,	0.00000798],
 275            [80,0.00005156,	0.00001571],
 276            ])
 277            
 278            #Are we dealing with z unit in meters or feet.
 279            unit_index = 1 #1: Meters, 2: feet
 280
 281            #Use the upper left coordinate of the raster to calculate z-value. A better option might be to
 282            # estimate the center of the raster, but that could be slightly more complicated if rasters were skewed
 283            uly  = rasterSrc.GetGeoTransform()[3] #Get the upper left y-coordinate of the raster
 284
 285            #Interpolate as needed
 286            zFactor = np.interp(uly,zFactorsTable[:,0],zFactorsTable[:,unit_index])
 287        else:
 288            print('Warning: non-standard units detected in geographic coordinate system, proceeding with a z_factor of 1')
 289
 290    return zFactor

 292def _apply_kernel_to_grid(grid:np.ndarray, kernel:np.ndarray, doUseConvolve:bool = True):
 293    """Convolves the supplied kernel with the grid. If doUseConvolve is True this applies the scipy.ndimage.convolve
 294    function, otherwise this uses scipy.ndimage.generic_filter - in some situations this may be faster.
 295
 296    Args:
 297        grid (numpy.ndarray): Numpy array of the digital elevation model of interest
 298        kernel (numpy.ndarray): Numpy array of the kernel to apply
 299        doUseConvolve (bool, optional): Should ndimage.convolve be used? If False, instead iterates
 300            using scipy.ndimage.generic_filter. Defaults to True.
 301
 302    Returns:
 303        outputGrid (numpy.ndarray): The result of the convolution
 304    """
 305    if doUseConvolve:
 306        #Preform the convolution
 307        outputGrid = ndi.convolve(grid,kernel,**NDICONVOLVE_KWARGS)
 308    else:
 309        #Preform as an iteration
 310        sub_fun = lambda subarray: np.sum(kernel.flatten()*subarray.flatten())
 311        outputGrid = ndi.generic_filter(grid,sub_fun,kernel.shape,**NDICONVOLVE_KWARGS)
 312
 313    return outputGrid

 315def _get_windowed_mean_square_array(rasterGrid:np.ndarray, windowRadius:int = 1,
 316                          doUseConvolve:bool = True):
 317    """Calculates a windowed mean in a square area, where the window is of size
 318    (2*windowRadius + 1,2*windowRadius + 1) so that it is always centered about
 319    the pixel of interest.
 320
 321    Args:
 322        rasterGrid (numpy.ndarray): Numpy array of the digital elevation model of interest
 323        windowRadius (int, optional): Defines the window shape as (2*windowRadius + 1,2*windowRadius + 1).
 324            Defaults to 1.
 325        doUseConvolve (bool, optional): Should ndimage.convolve be used? If False, instead iterates
 326            using scipy.ndimage.generic_filter. Defaults to True.
 327
 328    Returns:
 329        meanRaster (numpy.ndarray): The result of applying the square windowed mean filter to the input grid
 330    """
 331
 332    #build kernel
 333    meanKernel = np.ones((2*windowRadius + 1,2*windowRadius + 1))/((2*windowRadius + 1)**2)
 334
 335    #Preform the convolution
 336    meanRaster = _apply_kernel_to_grid(rasterGrid,meanKernel,doUseConvolve=doUseConvolve)
 337
 338    return meanRaster

 341def _get_windowed_mean_circular_array(rasterGrid:np.ndarray, windowRadius:int = 1,
 342                            doUseConvolve:bool = True):
 343    
 344    """Calculates a windowed mean in a circular area, where the window is of size
 345    (2*windowRadius + 1,2*windowRadius + 1) so that it is always centered about
 346    the pixel of interest.
 347
 348    Args:
 349        rasterGrid (numpy.ndarray): Numpy array of the digital elevation model of interest
 350        windowRadius (int, optional): Defines the window shape as (2*windowRadius + 1,2*windowRadius + 1).
 351            Defaults to 1.
 352        doUseConvolve (bool, optional): Should ndimage.convolve be used? If False, instead iterates
 353            using scipy.ndimage.generic_filter. Defaults to True.
 354
 355    Returns:
 356        meanRaster (numpy.ndarray): The result of applying the circular windowed mean filter to the input grid
 357    """
 358    #build a circular mean kernel
 359    X,Y = np.meshgrid(np.arange(-windowRadius,windowRadius+1), np.arange(-windowRadius,windowRadius+1)) #coordinates from center of kernel
 360    dist = np.sqrt(X**2 + Y**2) #distance from center of kernel
 361    meanKernel = 1.0*(dist <= windowRadius+0.5) #ones where cell center is at or less than radius
 362    meanKernel/=np.sum(meanKernel) #divide kernel by number of non-zero cells
 363
 364    #Preform the convolution
 365    meanRaster = _apply_kernel_to_grid(rasterGrid,meanKernel,doUseConvolve=doUseConvolve)
 366
 367    return meanRaster

 456def _get_gaussian_mean_array(rasterGrid:np.ndarray, sigma:float, truncateStds:float,
 457                             doUseConvolve:bool):
 458    """Calculates a windowed mean weighted by a gaussian distribution with a standard deviation
 459    (in pixel units) of sigma and a window size that extends radially out to truncateStds*sigma.
 460
 461    Args:
 462        rasterGrid (numpy.ndarray): Numpy array of the digital elevation model of interest
 463        sigma (float): standard deviation of gaussian distribution used to calculates weights for the
 464            weighted mean kernel.
 465        truncateStds (float): The number of standard deviations away from the central pixel that the weighted
 466            mean will be applied to. Creates a kernel that is of shape
 467            (2*int(sigma*truncateStds)+1, 2*int(sigma*truncateStds)+1)
 468        doUseConvolve (bool, optional): Should ndimage.convolve be used? If False, instead iterates
 469            using scipy.ndimage.generic_filter. Defaults to True.
 470
 471    Returns:
 472        gaussMeanRaster (numpy.ndarray): Grid representing the input rasterGrid after being smoothed by the 
 473            requested gaussian mean kernel.
 474        
 475    """
 476    #build a circular gaussian kernel
 477    radius = int(sigma*truncateStds)
 478
 479    X,Y = np.meshgrid(np.arange(-radius,radius+1), np.arange(-radius,radius+1)) #coordinates from center of kernel
 480    dist = np.sqrt(X**2 + Y**2) #distance from center of kernel
 481
 482    #Construct the gaussian kernel
 483    gauss_kernel = np.sqrt(2*np.pi*sigma**2)*np.exp(-0.5*((dist/sigma)**2))
 484    gauss_kernel/=np.sum(gauss_kernel)
 485
 486    #Preform the convolution
 487    gaussMeanRaster = _apply_kernel_to_grid(rasterGrid,gauss_kernel,doUseConvolve=doUseConvolve)
 488
 489    return gaussMeanRaster

 617def _get_slope_magnitude_array(rasterGrid:np.ndarray, nPixelRadius:int = 1,
 618                               dx:float = 1, dy:float=-1, zFactor:float = 1,
 619                               doUseConvolve:bool = True): 
 620    """Calculate a finite difference approximation of the magnitude of the slope on a
 621    2d numpy array. Uses a standard second-ordered centered approximation of slope if nPixelRadius is 1.
 622    In general, measures the slope by differencing the values of nPixelRadius in front of and behind each
 623    cell (e.g., dz_i/dx = (Z_i+nPixelRadius - Z_i-nPixelRadius)/(2*dx*nPixelRadius))
 624
 625    Args:
 626        rasterGrid (numpy.ndarray): The digital elevation model represented as a numpy array.
 627        nPixelRadius (int, optional): The distance away from the central pixel to use
 628            in the finite difference approximation. Larger values will average slope over
 629            a greater distance. Defaults to 1.
 630        dx (float, optional): The cell size in the x (e.g., easting) direction. Defaults to 1.
 631        dy (float, optional): The cell size in the y (e.g., northing) direction. Defaults to 1.

 636
 637    Returns:
 638        slopeMagArray (numpy.ndarray): The magnitude of slope estimated with the requested centered difference kernel.
 639    """
 640    #Transform dx,dy based on the zFactor
 641    dx/=zFactor
 642    dy/=zFactor
 643
 644    #Transform dx, dy based on the nPixelRadius
 645    dx*=2*nPixelRadius
 646    dy*=2*nPixelRadius
 647
 648    #Initialize the kernels in the x and y directions
 649    Sx_kernel = np.zeros((nPixelRadius*2 + 1, nPixelRadius*2 + 1))
 650    Sy_kernel = np.zeros_like(Sx_kernel)
 651    
 652    #create kernel and perform the convolution
 653    Sx_kernel[nPixelRadius,0] = -1/dx
 654    Sx_kernel[nPixelRadius,-1] = 1/dx

 656    #create kernel and perform the convolution
 657    Sy_kernel[0,nPixelRadius] = -1/dy
 658    Sy_kernel[-1,nPixelRadius] = 1/dy

 661    Sy=_apply_kernel_to_grid(rasterGrid,Sy_kernel,doUseConvolve)
 662
 663    #Calculate the magnitude
 664    slopeMagArray = np.sqrt(Sx**2 + Sy**2)
 665

 810                               doUseConvolve:bool = True):

 812    the central pixel and pixels nPixelRadius away in the x and y directions.
 813    
 814    For example, in one dimension d^2z_i/dx^2 = (2*z_i - z_(i-nPixelRadius) - z(i+nPixelRadius))/(dx*nPixelRadius)**2
 815

 899def _get_tpi_array(rasterGrid:np.ndarray, outerPixelRadius:int = 10, innerPixelRadius:int = 3,
 900                   distanceWeightingExponent:float = 0, doUseConvolve:bool = True):
 901    """Calculate the topographic position index, the difference between the digital elevation model
 902    and the mean calculated in an annulus centered around each pixel.
 903
 904    Args:
 905        rasterGrid (numpy.ndarray): The digital elevation model represented as a numpy array.
 906        outerPixelRadius (int, optional): The outer radius of the annulus, measured as a number of pixels.
 907            Defaults to 10.
 908        innerPixelRadius (int, optional): The inner radius of the annulus, measured as a number of pixels.
 909            Defaults to 3.
 910        distanceWeightingExponent (float, optional): The exponent to use in weighting values as a function
 911            of distance to calculate the annulus mean. Values less than 0 can be used to calculate a weighted mean
 912            that provides greater weight to pixels near the center of the annulus.
 913            Defaults to 0 (equivalent to no weighting by distance). 
 914        doUseConvolve (bool, optional): Should ndimage.convolve be used? If False, instead iterates
 915            using scipy.ndimage.generic_filter. Defaults to True.
 916
 917    Returns:
 918        tpiArray (np.ndarray): TPI calculated with the requested kernel.
 919    """
 920    #X,Y coordinates of the kernel from center
 921    X,Y = np.meshgrid(np.arange(-outerPixelRadius,outerPixelRadius+1),
 922                       np.arange(-outerPixelRadius,outerPixelRadius+1))
 923    
 924    #radial distance of kernel
 925    dist = np.sqrt(X**2 + Y**2)
 926
 927    #Mask the distance
 928    in_bounds = (dist < outerPixelRadius) & (dist >=innerPixelRadius)
 929    
 930    #calculate the kernel as the inverse of distance raised to an exponent
 931    #if that exponent is 0 (the default), weights will be 1 for all cells in donut
 932    kernel = in_bounds*(dist**-distanceWeightingExponent)
 933
 934    #normalize the kernel values
 935    kernel/=np.sum(kernel)
 936
 937    #Perform the convolution
 938    donut_mean = _apply_kernel_to_grid(rasterGrid,kernel,doUseConvolve=doUseConvolve)
 939
 940    #calculate difference the kernel mean and the local values
 941    tpiArray = rasterGrid - donut_mean
 942
 943    return tpiArray
